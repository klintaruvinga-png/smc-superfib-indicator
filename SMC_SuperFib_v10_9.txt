first_trading_day_of_month(_y, _m) =>
    t1  = timestamp("America/New_York", _y, _m, 1, 0, 0)
    dow = dayofweek(t1, "America/New_York")
    t1 + (dow == 7 ? 2 * 86400000 : dow == 1 ? 86400000 : 0)

// ── FORWARD ANCHORS ────────────────────────────────────────────────
next_sunday(t) =>
    int _result = int(na)
    if not na(t)
        dow         = dayofweek(t, "America/New_York")
        days_to_add = dow == 7 ? 0 : 7 - dow
        ts          = timestamp("America/New_York", year(t, "America/New_York"), month(t, "America/New_York"), dayofmonth(t, "America/New_York"), 0, 0)
        _result    := ts + days_to_add * 86400000
    _result

_anchor_mo(_yr, _mo, _off) =>
    _m = _mo + _off
    _y = _yr + (_m > 12 ? 1 : 0)
    _m2 = _m > 12 ? _m - 12 : _m
    first_trading_day_of_month(_y, _m2)

_anchor_qtr(_yr, _mo, _off) =>
    _q_start   = _mo <= 3 ? 1 : _mo <= 6 ? 4 : _mo <= 9 ? 7 : 10
    _target_mo = _q_start + _off * 3
    _extra_yrs = (_target_mo - 1) / 12
    _y         = _yr + _extra_yrs
    _m         = _target_mo - _extra_yrs * 12
    first_trading_day_of_month(_y, _m)

        bear_w  = math.pow(pos, 2.0)
        in_chop = pos >= 0.375 and pos <= 0.625
        [in_chop ? bull_w * 0.5 : bull_w, in_chop ? bear_w * 0.5 : bear_w]

cur_price_zone() =>
    ref_hi = not na(f3_high) and f3_high != 0.0 ? f3_high : not na(f1_high) ? f1_high : na
    ref_lo = not na(f3_low)  and f3_low  != 0.0 ? f3_low  : not na(f1_low)  ? f1_low  : na
    if na(ref_hi) or na(ref_lo) or ref_hi == ref_lo
        0
    else
        rng = ref_hi - ref_lo
        close >= ref_hi - rng * 0.375 ? 1 : close <= ref_hi - rng * 0.625 ? -1 : 0

sf_zone(p, hi, lo) =>
    rng = hi - lo
    p >= hi - rng*0.375 ? 1 : p <= hi - rng*0.625 ? -1 : 0

sf_side(p, hi, lo) =>
    rng = hi - lo
    mid = (hi + lo) / 2
    p >= mid ? 1 : -1

sf_key(hi, lo, idx) =>
    rng = hi - lo
    idx == 0  ? hi              :
     idx == 1  ? hi - rng*0.25  :
     idx == 2  ? hi - rng*0.50  :
     idx == 3  ? hi - rng*0.625 :
     idx == 4  ? hi - rng*0.75  :
     idx == 5  ? lo             :
     idx == 6  ? hi + rng*0.25  :
     idx == 7  ? hi + rng*0.50  :
     idx == 8  ? hi + rng*0.625 :

     idx == 10 ? hi + rng*1.00  :

     idx == 15 ? lo - rng*0.625 :

base_stars_at_close() =>
    ref_hi = not na(f3_high) and f3_high != 0.0 ? f3_high : not na(f1_high) ? f1_high : na
    ref_lo = not na(f3_low)  and f3_low  != 0.0 ? f3_low  : not na(f1_low)  ? f1_low  : na
    if na(ref_hi) or na(ref_lo) or ref_hi == ref_lo
        0
    else
        _have_f1 = not na(f1_high) and not na(f1_low)
        _have_f2 = not na(f2_high) and not na(f2_low)
        _have_f3 = not na(f3_high) and not na(f3_low)
        _ref_rng = _have_f3 ? (f3_high - f3_low) : _have_f1 ? (f1_high - f1_low) : (f2_high - f2_low)
        _w       = _ref_rng * (sf_tolerance / 100)
        _best    = 0
        for fi = 1 to 3
            fh = fi == 1 ? f1_high : fi == 2 ? f2_high : f3_high
            fl = fi == 1 ? f1_low  : fi == 2 ? f2_low  : f3_low
            _fa = fi == 1 ? _have_f1 : fi == 2 ? _have_f2 : _have_f3
            if _fa and not na(fh) and not na(fl)
                for ki = 0 to 19
                    pk = sf_key(fh, fl, ki)
                    zk = sf_zone(pk, fh, fl)
                    if zk != 0 and math.abs(close - pk) <= _w * 2
                        _count = 1
                        for fj = 1 to 3
                            if fj != fi
                                oh  = fj == 1 ? f1_high : fj == 2 ? f2_high : f3_high
                                ol  = fj == 1 ? f1_low  : fj == 2 ? f2_low  : f3_low
                                _fb = fj == 1 ? _have_f1 : fj == 2 ? _have_f2 : _have_f3
                                if _fb and not na(oh) and not na(ol)
                                    for oi = 0 to 19
                                        op = sf_key(oh, ol, oi)
                                        oz = sf_zone(op, oh, ol)
                                        if oz == zk and math.abs(pk - op) <= _w
                                            _count += 1
                                            break
                        if _count > _best
                            _best := _count
        _best

    ref_lo = not na(f3_low)  and f3_low  != 0.0 ? f3_low  : not na(f1_low)  ? f1_low  : na
    if na(ref_hi) or na(ref_lo) or ref_hi == ref_lo
        0
    else
        _best_lvl = float(na)
        _best_d   = float(na)
        for fi = 1 to 3
            fh = fi == 1 ? f1_high : fi == 2 ? f2_high : f3_high
            fl = fi == 1 ? f1_low  : fi == 2 ? f2_low  : f3_low
            _fa = false
            if fi == 1
                _fa := not na(f1_high) and not na(f1_low)
            else if fi == 2
                _fa := not na(f2_high) and not na(f2_low)
            else
                _fa := not na(f3_high) and not na(f3_low)
            if _fa and not na(fh) and not na(fl)
                for ki = 0 to 19
                    pk = sf_key(fh, fl, ki)
                    zk = sf_zone(pk, fh, fl)
                    if zk == pz
                        _d = math.abs(close - pk)
                        if na(_best_d) or _d < _best_d
                            _best_d   := _d
                            _best_lvl := pk
        if na(_best_lvl)
            0
        else
            origin     = pz == -1 ? last_ph1 : last_pl1
            if na(origin)
                0
            else
                total_dist = math.abs(origin - _best_lvl)
                if total_dist <= 0
                    0
                else
                    travelled = math.abs(origin - close)
                    ratio     = math.min(travelled / total_dist, 1.0)
                    ratio >= 0.66 ? 2 : ratio >= 0.33 ? 1 : 0

f_ede_tier(level_price, fib_high, fib_low) =>
    if fib_high == fib_low or na(fib_high) or na(fib_low)
        1
    else
        range_val = fib_high - fib_low
        price_pos = (level_price - fib_low) / range_val
        distance  = math.abs(price_pos - 0.5)
        distance < 0.125 ? 0 : distance < 0.4 ? 1 : distance < 0.75 ? 2 : distance < 1.25 ? 3 : distance < 2.0 ? 4 : 5

f_hybrid_stars(level_price, fib_high, fib_low, conf_count) =>
    base_tier  = f_ede_tier(level_price, fib_high, fib_low)
    conf_bonus = conf_count >= 3 ? 1.0 : conf_count == 2 ? 0.5 : 0.0
    raw        = base_tier + conf_bonus
    capped     = math.min(raw, base_tier + 1)
    math.min(math.round(capped), 5)

// PATCH K: HTF SL buffer scaling implemented inside get_sl_from_sweep
get_sl_from_sweep(direction, is_sweep_up, is_sweep_down, swing_hi, swing_lo) =>
    float sl = na
    buffer = pip_type == "JPY" ? 1.0 * pip_mult : 10.0 * pip_mult
    // Apply HTF scaling based on session_tf
    if session_tf == "Yearly" or session_tf == "Monthly" or session_tf == "Weekly"
        buffer := buffer * 2.0
    else if session_tf == "Daily"
        buffer := buffer * 1.5
    // Legacy fallback using original inputs (kept for compatibility)
    if session_tf == "Yearly"
        buffer := pip_type == "JPY" ? 2.0 * pip_mult : 20.0 * pip_mult
    else if session_tf == "Monthly"
        buffer := pip_type == "JPY" ? 1.5 * pip_mult : 15.0 * pip_mult
    if direction == "SELL" and is_sweep_up
        sl := swing_hi + buffer
    else if direction == "BUY" and is_sweep_down
        sl := swing_lo - buffer
    sl

get_tp_price_from_zone(zone_price, direction) =>
    float tp = na
    if direction == "SELL"
        levels = array.new_float()
        if not na(ef_l0625)  and ef_l0625  < zone_price - 0.00001
            array.push(levels, ef_l0625)
        if not na(ef_l075)   and ef_l075   < zone_price - 0.00001
            array.push(levels, ef_l075)
        if not na(ef_l100)   and ef_l100   < zone_price - 0.00001
            array.push(levels, ef_l100)
        if not na(f3_low)    and f3_low    < zone_price - 0.00001
            array.push(levels, f3_low)
        if array.size(levels) > 0
            array.sort(levels, order.descending)
            tp := array.get(levels, 0)
    else
        levels = array.new_float()
        if not na(ef_l000)  and ef_l000  > zone_price + 0.00001
            array.push(levels, ef_l000)
        if not na(ef_l025)  and ef_l025  > zone_price + 0.00001
            array.push(levels, ef_l025)
        if not na(ef_l0375) and ef_l0375 > zone_price + 0.00001
            array.push(levels, ef_l0375)
        if not na(f3_high)  and f3_high  > zone_price + 0.00001
            array.push(levels, f3_high)
        if array.size(levels) > 0
            array.sort(levels)
            tp := array.get(levels, 0)
    tp

compute_score(pz, bst) =>
    ef_al = 0
    if (pz == 1 and ef_near_entry_sell) or (pz == -1 and ef_near_entry_buy)
        ef_al := 2
    else
        sell_p = pz == 1 and (ef_dist_625 <= prox_thresh * 2 or ef_dist_750 <= prox_thresh * 2)
        buy_p  = pz == -1 and (ef_dist_375 <= prox_thresh * 2 or ef_dist_250 <= prox_thresh * 2)
        if sell_p or buy_p
            ef_al := 1
    ote_al = 0
    if pz == 1 and not na(ef_l0625) and not na(ef_l075) and close >= ef_l0625 and close <= ef_l075
        ote_al := 2
    else if pz == -1 and not na(ef_l025) and not na(ef_l0375) and close >= ef_l025 and close <= ef_l0375
        ote_al := 2
    sb = 0
    if (pz == 1 and ict_struct_trend == -1) or (pz == -1 and ict_struct_trend == 1)
        sb := 1
    else if ict_struct_trend != 0 and ((pz == 1 and ict_struct_trend != -1) or (pz == -1 and ict_struct_trend != 1))
        sb := -1
    bst + ef_al + ote_al + sb

calc_setup_quality(is_buy, major_ok, poi_ok, fvg_ok, poi_fresh_ok) =>
    int q = 0
    q += liquidity_type != "NONE" ? 20 : 0
    q += is_buy ? (mss_bullish ? 15 : 0) : (mss_bearish ? 15 : 0)
    q += major_ok ? 15 : 0
    q += poi_ok ? 20 : 0
    q += fvg_ok ? 10 : 0
    q += ((is_buy and ict_struct_trend == 1) or ((not is_buy) and ict_struct_trend == -1)) ? 10 : 0
    q += session_ok ? 10 : 0
    q += poi_fresh_ok ? 0 : -10
    math.max(0, math.min(q, 100))

calc_execution_quality(zone_price, is_buy, rr_ok) =>
    int q = 0
    zone_dist = not na(zone_price) ? math.abs(close - zone_price) * (pip_type == "JPY" ? 100.0 : 10000.0) : na
    q += not na(zone_dist) ? math.max(0, 25 - int(math.min(zone_dist, 25.0))) : 0
    q += ((is_buy and ef_near_entry_buy) or ((not is_buy) and ef_near_entry_sell)) ? 20 : 0
    q += rr_ok ? 25 : 0
    q += session_ok ? 10 : 0
    q += valid_sequence ? 10 : 0
    q += not na(zone_dist) and zone_dist <= 20 ? 10 : 0
    math.max(0, math.min(q, 100))

grade_setup(setup_q, exec_q, rr_ok, poi_ok, major_ok) =>
    rr_ok and poi_ok and major_ok and setup_q >= 85 and exec_q >= 70 ? "A+" :
     rr_ok and poi_ok and major_ok and setup_q >= 75 ? "A" :
     rr_ok and poi_ok and setup_q >= 60 ? "B" :
     rr_ok and setup_q >= 45 ? "C" : "BLOCKED"
