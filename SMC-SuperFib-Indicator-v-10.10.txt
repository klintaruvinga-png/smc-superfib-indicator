first_trading_day_of_month(_y, _m) =>
    t1  = timestamp("America/New_York", _y, _m, 1, 0, 0)
    dow = dayofweek(t1, "America/New_York")
    t1 + (dow == 7 ? 2 * 86400000 : dow == 1 ? 86400000 : 0)

// ── FORWARD ANCHORS ────────────────────────────────────────────────
next_sunday(t) =>
    int _result = int(na)
    if not na(t)
        dow         = dayofweek(t, "America/New_York")
        days_to_add = dow == 7 ? 0 : 7 - dow
        ts          = timestamp("America/New_York", year(t, "America/New_York"), month(t, "America/New_York"), dayofmonth(t, "America/New_York"), 0, 0)
        _result    := ts + days_to_add * 86400000
    _result

_anchor_mo(_yr, _mo, _off) =>
    _m = _mo + _off
    _y = _yr + (_m > 12 ? 1 : 0)
    _m2 = _m > 12 ? _m - 12 : _m
    first_trading_day_of_month(_y, _m2)

_anchor_qtr(_yr, _mo, _off) =>
    _q_start   = _mo <= 3 ? 1 : _mo <= 6 ? 4 : _mo <= 9 ? 7 : 10
    _target_mo = _q_start + _off * 3
    _extra_yrs = (_target_mo - 1) / 12
    _y         = _yr + _extra_yrs
    _m         = _target_mo - _extra_yrs * 12
    first_trading_day_of_month(_y, _m)

    if na(fib_hi) or na(fib_lo) or fib_hi == fib_lo

    else
        rng     = fib_hi - fib_lo
        pos     = math.max(0.0, math.min(1.0, (close - fib_lo) / rng))
        bull_w  = math.pow(1.0 - pos, 2.0)
        bear_w  = math.pow(pos, 2.0)
        in_chop = pos >= 0.375 and pos <= 0.625
        [in_chop ? bull_w * 0.5 : bull_w, in_chop ? bear_w * 0.5 : bear_w]

// ── DYNAMIC FIB LEVEL ARRAYS (v10.10 — FIX 2) ────────────────────
// Single source-of-truth: base levels + extension inputs
// Base:       0.0, 0.25, 0.5, 0.625, 0.75, 1.0
// Extensions: 0.625, 1.0, 1.625, 2.0
// Generated:  base + positive exts (1+ext) + negative exts (-ext)

var array<float> fib_ratios = array.new_float()

f_build_fib_levels() =>
    array<float> _levels = array.new_float()
    // Base levels (internal: 0 → 1.0)
    float[] _base = array.from(0.0, 0.25, 0.5, 0.625, 0.75, 1.0)
    for _b in _base
        if not array.includes(_levels, _b)
            array.push(_levels, _b)
    // Extension inputs
    float[] _exts = array.from(0.25, 0.625, 1.0, 1.625, 2.0)
    // Positive extensions (buy-side): 1 + ext
    for _e in _exts
        _pos = 1.0 + _e
        if not array.includes(_levels, _pos)
            array.push(_levels, _pos)
    // Negative extensions (sell-side): -ext
    for _e in _exts
        _neg = -_e
        if not array.includes(_levels, _neg)
            array.push(_levels, _neg)
    // Sort ascending
    array.sort(_levels, order.ascending)
    _levels

in_sb_c      = not na(time("15", "1400-1500", "America/New_York"))

    idx >= 0 and idx < array.size(fib_labels) ? array.get(fib_labels, idx) : "?"

// ── HYBRID EDE STAR ENGINE ────────────────────────────────────────
f_ede_tier(level_price, fib_high, fib_low) =>
    if fib_high == fib_low or na(fib_high) or na(fib_low)
        1
    else
        range_val = fib_high - fib_low
        price_pos = (level_price - fib_low) / range_val
        distance  = math.abs(price_pos - 0.5)
        distance < 0.125 ? 0 : distance < 0.4 ? 1 : distance < 0.75 ? 2 : distance < 1.25 ? 3 : distance < 2.0 ? 4 : 5

f_hybrid_stars(level_price, fib_high, fib_low, conf_count) =>
    base_tier  = f_ede_tier(level_price, fib_high, fib_low)
    conf_bonus = conf_count >= 3 ? 1.0 : conf_count == 2 ? 0.5 : 0.0
    raw        = base_tier + conf_bonus
    capped     = math.min(raw, base_tier + 1)
    math.min(math.round(capped), 5)

f_star_label(stars) =>
    stars == 0 ? "★0" : stars == 1 ? "★" : stars == 2 ? "★★" : stars == 3 ? "★★★" : stars == 4 ? "★★★★" : "★★★★★"

f_lvl(a, b, ratio) => a - (a - b) * ratio

draw_vert_arrow_t(x_t, hi, lo, point_up, vc, lw) =>
    wing_sz  = (hi - lo) * 0.15
    wing_w   = _ms_per_bar * 2
    tip_y    = point_up ? hi : lo
    base_y   = point_up ? lo : hi
    // shaft — stops short of tip so arrowhead stands out
    shaft_end = point_up ? hi - wing_sz : lo + wing_sz
    line.new(x_t, base_y, x_t, shaft_end, xloc=xloc.bar_time, color=vc, width=lw, style=line.style_solid)
    // arrowhead left wing
    line.new(x_t, tip_y, x_t - wing_w, tip_y + (point_up ? -wing_sz : wing_sz), xloc=xloc.bar_time, color=vc, width=lw + 1, style=line.style_solid)
    // arrowhead right wing
    line.new(x_t, tip_y, x_t + wing_w, tip_y + (point_up ? -wing_sz : wing_sz), xloc=xloc.bar_time, color=vc, width=lw + 1, style=line.style_solid)
    // base crossbar (tail)
    line.new(x_t - wing_w, base_y, x_t + wing_w, base_y, xloc=xloc.bar_time, color=vc, width=lw, style=line.style_solid)

draw_fib_t(x_t, hi, lo, is_swap, point_up, lw_base, fib_name) =>
    if not na(hi) and not na(lo) and hi != 0.0 and lo != 0.0 and hi != lo
        p_edge  = is_swap ? swap_edge : prem_edge
        p_key   = is_swap ? swap_key  : prem_key
        p_ext1  = is_swap ? swap_ext  : prem_ext1
        p_ext2  = is_swap ? swap_ext  : prem_ext2
        d_edge  = is_swap ? swap_edge : disc_edge
        d_key   = is_swap ? swap_key  : disc_key
        d_ext1  = is_swap ? swap_ext  : disc_ext1
        d_ext2  = is_swap ? swap_ext  : disc_ext2
        arr_col = is_swap ? swap_key : point_up ? ARR_UP : ARR_DOWN
        rng     = math.abs(hi - lo)

        // v10.10 FIX 2+4+7: iterate sorted dynamic fib_ratios array
        // Price = hi - rng * ratio (non-directional, matches sf_key)
        for _idx = 0 to array.size(fib_ratios) - 1
            _ratio = array.get(fib_ratios, _idx)
            _price = f_lvl(hi, lo, _ratio)
            _lbl   = array.get(fib_labels, _idx)
            // Classify level for color + depth
            _is_ext   = _ratio < 0.0 or _ratio > 1.0
            _is_base  = _ratio >= 0.0 and _ratio <= 1.0
            _is_prem  = _ratio <= 0.5
            _is_chop  = _ratio > 0.375 and _ratio < 0.625
            _is_50    = math.abs(_ratio - 0.5) < 0.001
            // Skip extensions if not shown
            if _is_ext and not show_ext
                continue
            // Determine color + depth
            lc = color.white
            bc = color.white
            tc = color.white
            depth = 1
            if _is_50
                lc := is_swap ? color.new(swap_edge,50) : color.new(color.gray,50)
                bc := color.new(color.gray,95)
                tc := is_swap ? color.new(swap_edge,30) : color.new(color.gray,30)
                depth := 1
                _lbl := "50% ✕"
            else if _is_chop and _is_base
                lc := is_swap ? swap_edge : C_CHOP_BNDRY
                bc := lc
                tc := is_swap ? color.new(swap_edge,10) : color.new(color.orange,10)
                depth := 1
                _lbl := _lbl + " ⛔"
            else if _is_ext and _ratio < 0.0
                // Sell-side extensions (above high)
                _far = _ratio <= -1.0
                lc := _far ? p_ext2 : p_ext1
                bc := lc
                tc := lc
                depth := _far ? 5 : 4
            else if _is_ext and _ratio > 1.0
                // Buy-side extensions (below low)
                _far = _ratio >= 2.0
                lc := _far ? d_ext2 : d_ext1
                bc := lc
                tc := lc
                depth := _far ? 5 : 4
            else if _is_prem
                // Premium zone (0%-50%)
                _is_edge = math.abs(_ratio) < 0.001 or math.abs(_ratio - 0.25) < 0.001
                lc := _is_edge ? (_ratio < 0.001 ? p_key : p_edge) : p_edge
                bc := lc
                tc := lc
                depth := _ratio < 0.001 ? 3 : 2
            else
                // Discount zone (50%-100%)
                _is_key = math.abs(_ratio - 0.625) < 0.001 or math.abs(_ratio - 1.0) < 0.001
                lc := _is_key ? d_key : d_edge
                bc := lc
                tc := lc
                depth := _is_key ? 3 : 2
            _bsz = _is_50 ? 0.0 : band_size
            draw_level_t(x_t, _price, _bsz, fib_name, _lbl, lc, bc, tc, depth)
        // Entry buffer boxes at high/low
        if show_bands
            _eb_hi = box.new(x_t - _back_extension, hi + rng*0.25, x_t, hi - rng*0.25, xloc=xloc.bar_time,
                     border_color=color.new(#FFD700, 55), bgcolor=C_ENTRY_BUFFER, border_width=1)

            _eb_lo = box.new(x_t - _back_extension, lo + rng*0.25, x_t, lo - rng*0.25, xloc=xloc.bar_time,
                     border_color=color.new(#FFD700, 55), bgcolor=C_ENTRY_BUFFER, border_width=1)
            array.push(_fib_boxes, _eb_lo)
        if show_chop
            draw_chop_fill_t(x_t, hi, lo)
        draw_vert_arrow_t(x_t, hi, lo, point_up, arr_col, lw_base)

// PATCH K: HTF SL buffer scaling implemented inside get_sl_from_sweep
get_sl_from_sweep(direction, is_sweep_up, is_sweep_down, swing_hi, swing_lo) =>
    float sl = na
    buffer = pip_type == "JPY" ? 1.0 * pip_mult : 10.0 * pip_mult
    // Apply HTF scaling based on session_tf
    if session_tf == "Yearly" or session_tf == "Monthly" or session_tf == "Weekly"
        buffer := buffer * 2.0
    else if session_tf == "Daily"
        buffer := buffer * 1.5
    // Legacy fallback using original inputs (kept for compatibility)
    if session_tf == "Yearly"
        buffer := pip_type == "JPY" ? 2.0 * pip_mult : 20.0 * pip_mult
    else if session_tf == "Monthly"
        buffer := pip_type == "JPY" ? 1.5 * pip_mult : 15.0 * pip_mult
    if direction == "SELL" and is_sweep_up
        sl := swing_hi + buffer
    else if direction == "BUY" and is_sweep_down
        sl := swing_lo - buffer
    sl

compute_score(pz, bst) =>
    ef_al = 0
    if (pz == 1 and ef_near_entry_sell) or (pz == -1 and ef_near_entry_buy)
        ef_al := 2
    else
        sell_p = pz == 1 and (ef_dist_ote_lo <= prox_thresh * 2 or ef_dist_sweet <= prox_thresh * 2)
        buy_p  = pz == -1 and (ef_dist_ote_hi <= prox_thresh * 2 or ef_dist_sweet <= prox_thresh * 2)
        if sell_p or buy_p
            ef_al := 1
    ote_al = 0
    if not na(ef_ote_lo) and not na(ef_ote_hi) and close >= ef_ote_lo and close <= ef_ote_hi
        ote_al := 2
    sb = 0
    if (pz == 1 and ict_struct_trend == -1) or (pz == -1 and ict_struct_trend == 1)
        sb := 1
    else if ict_struct_trend != 0 and ((pz == 1 and ict_struct_trend != -1) or (pz == -1 and ict_struct_trend != 1))
        sb := -1
    bst + ef_al + ote_al + sb

// v10.13.0: PDA hierarchy ranking — ICT/SMC ranked execution preference
pda_rank_bonus(liq_type) =>
    liq_type == "EQH" or liq_type == "EQL" ? 10 :
     liq_type == "PDH" or liq_type == "PDL" ? 8 :
     liq_type == "PWH" or liq_type == "PWL" ? 6 :
     liq_type == "BSL" or liq_type == "SSL" ? 3 : 0

calc_setup_quality(is_buy, major_ok, poi_ok, fvg_ok, poi_fresh_ok) =>
    int q = 0
    // v10.13.0: ranked PDA liquidity (replaces flat 20)
    q += liquidity_type != "NONE" ? (10 + pda_rank_bonus(liquidity_type)) : 0
    q += is_buy ? (mss_bullish ? 15 : 0) : (mss_bearish ? 15 : 0)
    q += major_ok ? 15 : 0
    q += poi_ok ? 20 : 0
    q += fvg_ok ? 10 : 0
    q += (poi_ok and fvg_ok) ? 5 : 0                                                                    // v10.13.0: PDA confluence bonus
    q += ((is_buy and ict_struct_trend == 1) or ((not is_buy) and ict_struct_trend == -1)) ? 10 : 0
    q += session_ok ? 10 : 0
    q += poi_fresh_ok ? 0 : -10
    // v10.13.0: HTF PDA nesting — alignment bonus / conflict penalty
    q += ((is_buy and htf_trend == 1) or ((not is_buy) and htf_trend == -1)) ? 15 : 0
    q += ((is_buy and htf_trend == -1) or ((not is_buy) and htf_trend == 1)) ? -15 : 0
    math.max(0, math.min(q, 100))

calc_execution_quality(zone_price, is_buy, rr_ok) =>
    int q = 0
    zone_dist = not na(zone_price) ? math.abs(close - zone_price) * (pip_type == "JPY" ? 100.0 : 10000.0) : na
    q += not na(zone_dist) ? math.max(0, 25 - int(math.min(zone_dist, 25.0))) : 0
    // v10.13.0: EF proximity gated by narrative anchor quality
    q += ((is_buy and ef_near_entry_buy) or ((not is_buy) and ef_near_entry_sell)) ? (ef_is_narrative ? 20 : 10) : 0
    q += rr_ok ? 25 : 0
    q += session_ok ? 10 : 0
    q += valid_sequence ? 10 : 0
    q += not na(zone_dist) and zone_dist <= 20 ? 10 : 0
    // v10.13.0: Displacement / reaction quality filter
    q += bars_in_zone > 0 ? (bars_in_zone <= 3 ? 10 : bars_in_zone <= 6 ? 0 : -15) : 0
    q += reaction_candle_quality ? 10 : 0
    math.max(0, math.min(q, 100))

     rr_ok and poi_ok and setup_q >= 60 ? "B" :
     rr_ok and setup_q >= 45 ? "C" : "BLOCKED"

                float _sum_hi = 0.0
                float _sum_lo = 0.0

                if _sf_f1

                    _sum_lo := _sum_lo + f1_low

                    _sum_lo := _sum_lo + f2_low

            if _sf_valid and not na(_sf_anchor_hi) and not na(_sf_anchor_lo) and _sf_anchor_hi > _sf_anchor_lo

                for _si = 0 to _sf_max
                    if math.abs(array.get(fib_ratios, _si) - 0.5) < 0.001

                int _mono_buy  = 0

                // Sell-side: indices (_idx50-1) → 0  (ratio < 0.5, price > midpoint)
                // Walk closest-to-50% first so running-max enforces outward monotonicity
                if _idx50 > 0

                            _sf_star_raw            = f_ede_tier(_lp, _sf_anchor_hi, _sf_anchor_lo)
                            _sf_star_bonus          = math.max(_valid_count - 2, 0)
                            _sf_star_final_pre_mono = _sf_star_raw + _sf_star_bonus
                            _sf_star_final          = math.max(_sf_star_final_pre_mono, _mono_sell)

                            _ctier                  = _sf_star_final <= 1 ? 1 : _sf_star_final <= 3 ? 2 : 3
                            _ctier                  := math.min(_ctier, _valid_count)
                            if not sf_is_dup(_lp, _half)
                                sf_draw_t(_lp, _half, _ctier, _sf_star_final, _zone, _x_sf, _si)
                                array.push(_sf_drawn, _lp)

                // Buy-side: indices (_idx50+1) → _sf_max  (ratio > 0.5, price < midpoint)
                if _idx50 < _sf_max
                    for _bi = _idx50 + 1 to _sf_max
                        _lp = sf_key(_sf_anchor_hi, _sf_anchor_lo, _bi)

                            _sf_star_raw            = f_ede_tier(_lp, _sf_anchor_hi, _sf_anchor_lo)

                            _sf_star_final_pre_mono = _sf_star_raw + _sf_star_bonus
                            _sf_star_final          = math.max(_sf_star_final_pre_mono, _mono_buy)
                            _sf_star_final          := math.max(0, math.min(_sf_star_final, 5))
                            _mono_buy               := _sf_star_final
                            _ctier                  = _sf_star_final <= 1 ? 1 : _sf_star_final <= 3 ? 2 : 3
                            _ctier                  := math.min(_ctier, _valid_count)
                            if not sf_is_dup(_lp, _half)
                                sf_draw_t(_lp, _half, _ctier, _sf_star_final, _zone, _x_sf, _bi)
                                array.push(_sf_drawn, _lp)

                // Equilibrium (50%) — hardwired ★0 baseline, no bonus path
                _lp50 = sf_key(_sf_anchor_hi, _sf_anchor_lo, _idx50)
                if not na(_lp50) and not sf_is_dup(_lp50, _half)
                    _sf_star_raw            = 0
                    _sf_star_bonus          = 0
                    _sf_star_final_pre_mono = 0
                    _sf_star_final          = 0
